% --- Style commands ---
\newcommand{\sectionstyle}[1]{{\fontsize{14pt}{1em}\selectfont\bfseries\color{text}{\color{awesome}#1}}}
\newcommand{\entrytitlestyle}[1]{{\fontsize{10pt}{1em}\selectfont\bfseries\color{darktext} #1}}
\newcommand{\entrypositionstyle}[1]{{\fontsize{9pt}{1em}\selectfont\scshape\color{graytext} #1}}
\newcommand{\entrydatestyle}[1]{{\fontsize{9pt}{1em}\lightfont\slshape\color{graytext} #1}}
\newcommand{\entrylocationstyle}[1]{{\fontsize{9pt}{1em}\lightfont\slshape\color{awesome} #1}}
\newcommand{\descriptionstyle}[1]{{\fontsize{9pt}{1em}\lightfont\color{text} #1}}
\newcommand{\skilltypestyle}[1]{{\fontsize{10pt}{1em}\selectfont\bfseries\color{darktext} #1}}
\newcommand{\skillsetstyle}[1]{{\fontsize{9pt}{1em}\lightfont\color{text} #1}}

% --- Section command ---
\newcommand{\cvsection}[1]{%
  \vspace{6pt}%
  \sectionstyle{#1}%
  \phantomsection%
  \color{lighttext}\hrulefill%
  \vspace{4pt}%
}

% --- Entry command ---
% \cventry{position}{title}{location}{date}{description}
\newcommand{\cventry}[5]{%
  \vspace{-1mm}
  \setlength\tabcolsep{0pt}
  \begin{tabular*}{\textwidth}{@{\extracolsep{\fill}} L{\textwidth - 4.5cm} R{4.5cm}}
    \entrytitlestyle{#2} & \entrylocationstyle{#3} \\
    \entrypositionstyle{#1} & \entrydatestyle{#4} \\
  \end{tabular*}
  \descriptionstyle{#5}
}

% --- Skill table ---
\newenvironment{cvskills}{%
  \vspace{2pt}
  \begin{center}
    \setlength\tabcolsep{1ex}
    \begin{tabular*}{\textwidth}{@{\extracolsep{\fill}} r L{\textwidth * \real{0.88}}}
}{%
    \end{tabular*}
  \end{center}
}
\newcommand{\cvskill}[2]{%
  \skilltypestyle{#1} & \skillsetstyle{#2} \\
}

% --- Item list ---
\newenvironment{cvitems}{%
  \vspace{-3mm}
  \begin{itemize}[leftmargin=2ex, nosep, noitemsep]
    \renewcommand{\labelitemi}{\color{awesome}\textbullet}
    \setlength{\parskip}{0pt}
}{%
  \end{itemize}
  \vspace{-3mm}
}
